164 float ScaleValue(float baseValue, float scaleFactor, int level, float minValue, float maxValue, bool inverse = false)
165 {
166 // A skálázott érték kiszámítása, figyelembe véve az 'inverse' paramétert
167 float scaledValue = inverse
168 ? baseValue / (float)Mathf.Pow(scaleFactor, level - 1) // Fordított skálázás
169 : baseValue * (float)Mathf.Pow(scaleFactor, level - 1); // Normál skálázás
170
171 // Az érték korlátozása a megadott minimum és maximum határok között
172 return Mathf.Clamp(scaledValue, minValue, maxValue);
173 }
