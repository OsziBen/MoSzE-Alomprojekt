92 public GameObject GetProjectile(Vector2 position, Quaternion rotation, float damageValue, float percentageDMGValue)
93 {
94 // Loop, hogy megnézze van-e inaktív lövedék a pool-ban
95 foreach (var projectile in pool)
96 {
97 if (!projectile.activeInHierarchy)
98 {
99 // Ha találunk inaktív lövedéket, beállítjuk annak pozícióját és forgatását
100 projectile.transform.position = position;
101 projectile.transform.rotation = rotation;
102
103 // A lövedék tulajdonságainak frissítése (sebzés érték, módosító)
104 UpdateProjectileProperties(projectile, damageValue, percentageDMGValue);
105
106 // A lövedék aktiválása a pool-ból
107 projectile.SetActive(true);
108 // Az esemény triggerelése, hogy a lövedék aktiválva lett
109 OnProjectileActivated?.Invoke(projectile);
110 // Visszaadjuk a pool-ból lekért lövedéket
111 return projectile;
112 }
113 }
114
115 // Ha nincs elérhető inaktív lövedék, létrehozunk egy újat
116 GameObject newProjectile = Instantiate(projectilePrefab, position, rotation);
117 // Az új lövedék hozzáadása a pool-hoz
118 pool.Enqueue(newProjectile);
119
120 // Az új lövedék tulajdonságainak beállítása
121 UpdateProjectileProperties(newProjectile, damageValue, percentageDMGValue);
122
123 // Az esemény triggerelése az új lövedék aktiválásához
124 OnProjectileActivated?.Invoke(newProjectile);
125

164 void MarkProjectile(Projectile projectile)
165 {
166 // Ellenőrzi, hogy a lövések száma elérte-e a beállított intervallumot (projectileMarkEveryNth)
167 if (fireCounter % projectileMarkEveryNth == 0)
168 {
169 // Ha elérte az intervallumot, a lövedék jelölve lesz
170 projectile.IsMarked = true;
171 // A tüzelés számláló visszaállítása nullára, hogy újrainduljon az intervallum
172 fireCounter = 0;
173 }
174 else
175 {
176 // Ha nem érte el az intervallumot, a lövedék nem lesz jelölve
177 projectile.IsMarked = false;
178 }
