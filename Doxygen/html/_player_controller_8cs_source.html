185 void ApplyPlayerUpgradeStats(List<StatValuePair> playerUpgradeStatValues, float currentHealthPercentage)
186 {
187 // Biztosítja, hogy a currentHealthPercentage értéke 0 és 1 közé legyen korlátozva.
188 currentHealthPercentage = Mathf.Clamp01(currentHealthPercentage);
189
190 // Végigiterál minden egyes staton az upgrade listában.
191 foreach (var statValue in playerUpgradeStatValues)
192 {
193 // Ellenőrzés az adott upgrade típus alapján.
194 switch (statValue.Key)
195 {
196 case StatType.Health:
197 Debug.Log(statValue.Key + " :: " + statValue.Value);
198 // Növeli az additionalHealth értékét az adott frissítés alapján.
199 additionalHealth += statValue.Value;
200 // Az új maximális életerő értékének kiszámítása.
201 MaxHealth = ClampStat(maxHealth, additionalHealth, minHealthValue, maxHealthValue);
202 // Az aktuális életerő frissítése a százalékos egészség alapján.
203 CurrentHealth = MaxHealth * currentHealthPercentage;
204 break;
205 case StatType.MovementSpeed:
206 Debug.Log(statValue.Key + " :: " + statValue.Value);
207 additionalMovementSpeed += statValue.Value;
208 // A mozgási sebesség frissítése az új bónusz értékkel.
209 CurrentMovementSpeed = ClampStat(baseMovementSpeed, additionalMovementSpeed, minMovementSpeedValue, maxMovementSpeedValue);
210 break;
211 case StatType.Damage:
212 Debug.Log(statValue.Key + " :: " + statValue.Value);
213 additionalDMG += statValue.Value;
214 // A sebzés frissítése az új értékekkel.
215 CurrentDMG = ClampStat(baseDMG, additionalDMG, minDMGValue, maxDMGValue);
216 break;
217 case StatType.AttackCooldownReduction:
218 Debug.Log(statValue.Key + " :: " + statValue.Value);
219 attackCooldownReduction += statValue.Value;
220 // A támadás delay frissítése.
221 CurrentAttackCooldown = ClampStat(baseAttackCooldown, attackCooldownReduction, minAttackCooldownValue, maxAttackCooldownValue);
222 break;
223 case StatType.CriticalHitChance:
224 Debug.Log(statValue.Key + " :: " + statValue.Value);
225 additionalCriticalHitChance += statValue.Value;
226 // A kritikus találati esély frissítése.
227 CurrentCriticalHitChance = ClampStat(baseCriticalHitChance, additionalCriticalHitChance, minCriticalHitChanceValue, maxCriticalHitChanceValue);
228 break;
229 case StatType.PercentageBasedDMG:
230 Debug.Log(statValue.Key + " :: " + statValue.Value);
231 additionalPercentageBasedDMG += statValue.Value;
232 // A százalékos sebzés frissítése.
233 CurrentPercentageBasedDMG = ClampStat(basePercentageBasedDMG, additionalPercentageBasedDMG, minPercentageBasedDMGValue, maxPercentageBasedDMGValue);
234 break;
235 default:
236 // Ha egy nem kezelt stat típus érkezik, figyelmeztetést küld.
237 Debug.LogWarning($"Unhandled stat type: {statValue.Key}");
238 break;
239 }
240 }

310 void Update()
311 {
312 // A mozgási bemenet lekérése.
313 move = MoveAction.ReadValue<Vector2>();
314 // Ha van érvényes mozgási bemenet.
315 if (move != Vector2.zero)
316 {
317 // A játékos forgatása a mozgás irányának megfelelően.
318 // Az Atan2 kiszámítja az irány szögét radiánban, amit aztán fokokra konvertálunk.
319 // Az "-90f" érték azért van, hogy a sprite megfelelően igazodjon az irányhoz.
320 float angle = Mathf.Atan2(move.y, move.x) * Mathf.Rad2Deg;
321 transform.rotation = Quaternion.Euler(0f, 0f, angle -90f);
322 }
323 else
324 {
325 // Ha nincs mozgási bemenet, a játékos visszatér az alaphelyzetbe.
326 transform.rotation = Quaternion.Euler(0f, 0f, 0f);
327 }
328
329 // Ha a játékos sebezhetetlen, csökkenti a hátralévő időt.
330 if (isInvincible)
331 {
332 damageCooldown -= Time.deltaTime;
333 // Ha a cooldown idő lejárt, a sebezhetetlenség véget ér.
334 if (damageCooldown < 0)
335 {
336 isInvincible = false;
337 }
338 }
339
340 // Ha a játékos nem képes támadni, csökkenti a támadási cooldown időt.
341 if (!isAbleToAttack)
342 {
343 remainingAttackCooldown -= Time.deltaTime;
344 //Debug.Log(remainingAttackCooldown);
345
346 // Ha a cooldown idő lejárt, a játékos ismét támadhat.
347 if (remainingAttackCooldown <= 0)
348 {
349 isAbleToAttack = true;
350 }
351 }

360 void FixedUpdate()
361 {
362 Vector2 position = (Vector2)rigidbody2d.position + move * CurrentMovementSpeed * Time.deltaTime;
363
364 // Ellenõrizd, hogy az új pozíció kívül van-e a tartományon
365 if (position.x < movementBoundsMin.x || position.x > movementBoundsMax.x)
366 {
367 // Ha az X komponens kívül van a határokon, állítsd vissza az eredeti értékre
368 position.x = Mathf.Clamp(position.x, movementBoundsMin.x, movementBoundsMax.x);
369 }
370
371 if (position.y < movementBoundsMin.y || position.y > movementBoundsMax.y)
372 {
373 // Ha az Y komponens kívül van a határokon, állítsd vissza az eredeti értékre
374 position.y = Mathf.Clamp(position.y, movementBoundsMin.y, movementBoundsMax.y);
375 }
376
377 rigidbody2d.MovePosition(position);

409 void Attack(InputAction.CallbackContext context)
410 {
411 // Ellenőrzi, hogy a rigidbody vagy a támadás le van-e tiltva. Ha igen, azonnal visszatér.
412 if (rigidbody2d == null || !isAbleToAttack) return;
413 // A támadás helyi eltolásának meghatározása a bemenet alapján.
414 Vector2 localOffset = GetLocalOffset(context.control.name);
415 // Ha a bemenet érvénytelen (nulla eltolás), a metódus visszatér.
416 if (localOffset == Vector2.zero) return;
417
418 // Kiszámítja a támadási irányt egy szorzó segítségével, amely az eltolást méretezi.
419 float multiplier = GetMultiplier(localOffset);
420 attackDirection = rigidbody2d.position + localOffset * multiplier;
421
422 // Projectile objektum lekérése az objectpoolból.
423 GameObject projectileObject = GetProjectileFromPool();
424 // Ha nincs elérhető projektil, a metódus visszatér.
425 if (projectileObject == null) return;
426
427 // A projectile elindítása a megadott irányba.
428 LaunchProjectile(projectileObject);
429 // A támadási cooldown visszaállítása.
430 ResetAttackCooldown();
431 // A támadási képesség letiltása a cooldown idejére.
432 isAbleToAttack = false;
433 }

468 private float GetMultiplier(Vector2 localOffset)
469 {
470 // A karakter pillanatnyi iránya, amit a "transform.up" ad vissza
471 Vector2 facingDirection = transform.up;
472
473 // Kezeli a diagonális irányokat (pl. balra-fel vagy jobbra-le)
474 if (Mathf.Abs(facingDirection.x) > 0.1f && Mathf.Abs(facingDirection.y) > 0.1f)
475 {
476 // Ha a karakter diagonálisan néz, akkor az iránytól függően kombinált szorzót ad vissza
477 return (2f + 1f) / 2f; // A diagonális irányokhoz egy átlagos szorzót (1.5f) ad vissza
478 }
479
480 // Kezeli a vízszintes és függőleges irányokat
481 bool isHorizontal = Mathf.Abs(facingDirection.x) > Mathf.Abs(facingDirection.y);
482
483 // Ha a karakter vízszintes irányba néz
484 if (isHorizontal)
485 {
486 // A balra vagy jobbra irányban végzett támadásokhoz nagyobb szorzót (2f) ad, míg a függőleges irányú támadásokhoz 1f-et
487 return (localOffset == Vector2.left || localOffset == Vector2.right) ? 2f : 1f; // 2f for horizontal, 1f for vertical
488 }
489 else
490 {
491 // Ha a karakter függőleges irányba néz
492 // A fel és le irányú támadásokhoz nagyobb szorzót (2f) ad, míg a vízszintes irányú támadásokhoz 1f-et
493 return (localOffset == Vector2.up || localOffset == Vector2.down) ? 2f : 1f; // 2f for vertical, 1f for horizontal
494 }
495 }

607 OnPlayerDeath?.Invoke();
608 // A játékos GameObject-jét törli, így a karakter eltűnik a játékból
