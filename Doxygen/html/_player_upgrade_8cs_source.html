138 private string ReplacePlaceholders(string inputDescription)
139 {
140 // Általános helyõrzõk cseréje a megfelelõ értékekre.
141 inputDescription = inputDescription.Replace(@"\name", upgradeName); // A fejlesztés neve
142 inputDescription = inputDescription.Replace(@"\minUpgradeLevel", minUpgradeLevel.ToString()); // Minimális fejlesztési szint
143 inputDescription = inputDescription.Replace(@"\maxUpgradeLevel", maxUpgradeLevel.ToString()); // Maximális fejlesztési szint
144 inputDescription = inputDescription.Replace(@"\currentUpgradeLevel", currentUpgradeLevel.ToString()); // Jelenlegi fejlesztési szint
145 inputDescription = inputDescription.Replace(@"\basePrice", basePrice.ToString()); // Alapár
146 inputDescription = inputDescription.Replace(@"\priceScaleFactor", priceScaleFactor.ToString()); // Ár skálázási faktor
147
148 // A fejlesztés árának kiszámítása, figyelembe véve a szintet és az árskálázási tényezõt
149 int calculatedPriceValue = GetPrice();
150 inputDescription = inputDescription.Replace($@"\price", calculatedPriceValue.ToString()); // Ár helyõrzõ cseréje
151
152 // Dinamikusan a módosítók helyõrzõinek cseréje, ha van módosító
153 if (modifiers != null && modifiers.Count > 0)
154 {
155 for (int i = 0; i < modifiers.Count; i++)
156 {
157 // Módosító típusának, alapértékének és skálázási tényezõjének cseréje
158 inputDescription = inputDescription.Replace($@"\m[{i}].type", modifiers[i].type.ToString());
159 inputDescription = inputDescription.Replace($@"\m[{i}].baseValue", modifiers[i].baseValue.ToString());
160 inputDescription = inputDescription.Replace($@"\m[{i}].scaleFactor", modifiers[i].scaleFactor.ToString());
161
162 // Módosító értékének kiszámítása a szinttõl függõen
163 float calculatedModifierValue = modifiers[i].baseValue * Mathf.Pow(modifiers[i].scaleFactor, currentUpgradeLevel - minUpgradeLevel);
164 inputDescription = inputDescription.Replace($@"\m[{i}].value", calculatedModifierValue.ToString("F2")); // Módosító érték helyõrzõ cseréje
165 }
166 }
167
168 return inputDescription; // A végleges, helyettesített szöveg visszaadása
169 }

223 public int GetPrice()
224 {
225 // Az ár kiszámítása: alapár * (árskálázási faktor)^(jelenlegi szint - minimális szint)
226 // Az eredmény korlátozása a `basePrice` és `maxPrice` (150) közötti tartományra.
